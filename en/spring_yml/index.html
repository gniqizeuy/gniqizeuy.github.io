@Component
@Lombok
@ConfigurationProperties(prefix = "student")

}

Environment

    url: xxxxxxx

@Resource
private Environment env;

@GetMapping("/mes")
public String getMesUrl() {
    String url = env.getProperty("mes.url");
    
    return url;
}
