读取方式

  • @Value
  server:
    port: 8080

java

  @Value("${server.port}")
  private String port
  • ConfigurationProperties
student:
    name: jackma
    age: 18

javabean（需prefix (前缀) 参数）

@Component
@Lombok
@ConfigurationProperties(prefix = "student")
  public class Student {
      private String name;
  
      private Integer age;
  }
  • Environment
  mes:
    url: xxxxxxx

java

  @Resource
  private Environment env;
  
  @GetMapping("/mes")
  public String getMesUrl() {
      String url = env.getProperty("mes.url");
      
      return url;
  }
